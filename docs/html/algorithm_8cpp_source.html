285 static inline std::string base64EncodeImpl(std::span<const uint8_t> bytes,
286 const char* alphabet,
287 bool with_padding)
288 {
289 std::string out;
290 out.reserve(((bytes.size() + 2) / 3) * 4);
291 uint32_t val = 0;
292 int valb = -6;
293 for (uint8_t c : bytes) {
294 val = (val << 8) | c;
295 valb += 8;
296 while (valb >= 0) {
297 out.push_back(alphabet[(val >> valb) & 0x3F]);
298 valb -= 6;
299 }
300 }
301 if (valb > -6)
302 out.push_back(alphabet[((val << 8) >> (valb + 8)) & 0x3F]);
