159 template<typename T>
160 static nlohmann::json SerializeField(const T& value) {
161 if constexpr (std::is_base_of<SerializableBase, T>::value) {
162 return value.Serialize();
163 } else if constexpr (is_vector<T>::value || is_array<T>::value) {
164 nlohmann::json arr = nlohmann::json::array();
165 for (const auto& el : value) {

167 }
168 return arr;
169 } else if constexpr (is_map<T>::value || is_unordered_map<T>::value) {
170 nlohmann::json obj = nlohmann::json::object();
171 for (const auto& [k, v] : value) {
172 obj[k] = SerializeField(v);
173 }
174 return obj;
175 } else if constexpr (is_optional<T>::value) {
176 return value.has_value() ? SerializeField(*value) : nlohmann::json(nullptr);
177 } else if constexpr (is_unique_ptr<T>::value) {
178 return value ? SerializeField(*value) : nlohmann::json(nullptr);

193 template<typename T>
194 static void DeserializeField(T& field, const nlohmann::json& j) {
195 if constexpr (std::is_base_of<SerializableBase, T>::value) {
196 field.Deserialize(j);
197 } else if constexpr (is_vector<T>::value) {
198 field.clear();
199 for (const auto& el : j) {
200 typename T::value_type tmp;
201 DeserializeField(tmp, el);
202 field.push_back(std::move(tmp));
203 }
204 } else if constexpr (is_array<T>::value) {
205 size_t idx = 0;
206 for (const auto& el : j) {
207 DeserializeField(field[idx++], el);
208 }
209 } else if constexpr (is_map<T>::value || is_unordered_map<T>::value) {
210 field.clear();
211 for (auto it = j.begin(); it != j.end(); ++it) {
212 typename T::mapped_type tmp;
213 DeserializeField(tmp, it.value());
214 field[it.key()] = std::move(tmp);
215 }
216 } else if constexpr (is_optional<T>::value) {
217 if (j.is_null()) field.reset();
218 else {
219 typename T::value_type tmp;
220 DeserializeField(tmp, j);
221 field = std::move(tmp);
222 }
223 } else if constexpr (is_unique_ptr<T>::value) {
224 if (j.is_null()) field.reset();
225 else {
226 field = std::make_unique<typename T::element_type>();
227 DeserializeField(*field, j);
228 }
229 } else {
230 field = j.get<T>();
231 }
